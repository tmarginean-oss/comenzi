            <div id="listaCatalog" class="grid-produse"></div>

        <button class="btn-verde" onclick="finalizareComandaCompleta()">📋 Finalizează și Salvează în Jurnal</button>

        <p style="font-size: 14px; color: #6b46c1; margin-top: -5px;">Căutare rapidă pentru istoricul zilei sau comenzi de acum 18 zile.</p>
        <input type="text" id="cautaInJurnal" oninput="filtreazaJurnal()" placeholder="🔍 Caută după nume client sau dată (Ex: 06.05)...">

<script>
    // BAZA DE DATE OFICIALĂ DIN FIȘA TA DE PREȚURI
    const catalogMarfa = [
        {
            id: "rendes_vous", nume: "CUTII PATRATE CU CAPAC RENDES VOUS",
            variante: [
                { dim: "18*18*8", pret: 1.5 }, { dim: "20*20*8", pret: 1.7 }, { dim: "22*22*8", pret: 1.9 },
                { dim: "25*25*8", pret: 2.2 }, { dim: "28*28*8", pret: 2.8 }, { dim: "32*32*8", pret: 3.5 },
                { dim: "20*20*10", pret: 2.6 }, { dim: "23*23*10", pret: 2.9 }, { dim: "26*26*10", pret: 3.3 },
                { dim: "29*29*10", pret: 3.6 }, { dim: "32*32*10", pret: 4.1 }
            ]
        },
        {
            id: "gala", nume: "CUTII DREPTUNGHIULARE CU FEREASTRA GALA",
            variante: [
                { dim: "17*11.5*5", pret: 1.5 }, { dim: "17*14*6", pret: 1.7 },
                { dim: "26*18*8", pret: 2.5 }, { dim: "30*21*8", pret: 3.3 }
            ]
        },
        {
            id: "amore", nume: "CUTII PATRATE CU FEREASTRA AMORE",
            variante: [{ dim: "20*20*8", pret: 2.3 }]
        },
        {
            id: "haute_vision", nume: "CUTII HAUTE VISION",
            variante: [
                { dim: "25*25*5", pret: 2.6 }, { dim: "28*28*5", pret: 2.9 }, { dim: "20*10.5*5", pret: 1.5 }
            ]
        },
        {
            id: "prestige", nume: "CUTII DREPTUNGHIULARE PRESTIGE",
            variante: [
                { dim: "32*32*13", pret: 5.8 }, { dim: "40*40*13", pret: 7.8 },
                { dim: "42.5*28.5*6", pret: 4.9 }, { dim: "44*30*12", pret: 6.4 }, { dim: "62*42*13", pret: 13.6 }
            ]
        }
    ];

    let listaClientiObj = JSON.parse(localStorage.getItem('b2b_clienti_v11')) || [];
    let istoricPreturiClienti = JSON.parse(localStorage.getItem('b2b_istoric_v11')) || {};
    let jurnalComenziTrimise = JSON.parse(localStorage.getItem('b2b_jurnal_v11')) || [];

    let clientCurent = ""; let comandaCurenta = {}; let textGlobalRezumat = "";

    // Curățare automată la 365 de zile
    const UN_AN_IN_MS = 365 * 24 * 60 * 60 * 1000;
    jurnalComenziTrimise = jurnalComenziTrimise.filter(c => (Date.now() - c.timestamp) < UN_AN_IN_MS);
    localStorage.setItem('b2b_jurnal_v11', JSON.stringify(jurnalComenziTrimise));

    actualizeazaDropdownClienti();
    randareJurnalComenzi();

    function salveazaClientNou() {
        let nume = document.getElementById("numeClientNou").value.trim();
        let tel = document.getElementById("telefonClientNou").value.trim();
        let adr = document.getElementById("adresaClientNou").value.trim();
        if (!nume) { alert("Numele este obligatoriu!"); return; }
        if (listaClientiObj.some(c => c.nume.toLowerCase() === nume.toLowerCase())) { alert("Clientul există deja!"); return; }

        listaClientiObj.push({ nume: nume, telefon: tel, adresa: adr });
        istoricPreturiClienti[nume] = {};
        salveazaInMemorie(); actualizeazaDropdownClienti();
        document.getElementById("numeClientNou").value = "";
        document.getElementById("telefonClientNou").value = "";
        document.getElementById("adresaClientNou").value = "";
        alert(`Clientul "${nume}" a fost adăugat.`);
    }

    function stergeClientulSelectat() {
        let deSters = document.getElementById("selectClient").value;
        if (!deSters) { alert("Selectează un client!"); return; }
        if (confirm(`Ștergi definitiv clientul "${deSters}"?`)) {
            listaClientiObj = listaClientiObj.filter(c => c.nume !== deSters);
            delete istoricPreturiClienti[deSters];
            salveazaInMemorie(); actualizeazaDropdownClienti();
            document.getElementById("zonaProduse").style.display = "none";
            document.getElementById("zonaRezumat").style.display = "none";
        }
    }

    function actualizeazaDropdownClienti() {
        let select = document.getElementById("selectClient");
        select.innerHTML = '<option value="">-- Deschide lista de clienți --</option>';
        listaClientiObj.sort((a,b) => a.nume.localeCompare(b.nume)).forEach(c => {
            select.innerHTML += `<option value="${c.nume}">${c.nume}</option>`;
        });
    }

    function salveazaInMemorie() {
        localStorage.setItem('b2b_clienti_v11', JSON.stringify(listaClientiObj));
        localStorage.setItem('b2b_istoric_v11', JSON.stringify(istoricPreturiClienti));
        localStorage.setItem('b2b_jurnal_v11', JSON.stringify(jurnalComenziTrimise));
    }

    function incarcaClient() {
        clientCurent = document.getElementById("selectClient").value;
        if (!clientCurent) { document.getElementById("zonaProduse").style.display = "none"; document.getElementById("zonaRezumat").style.display = "none"; return; }
        let dateClient = listaClientiObj.find(c => c.nume === clientCurent);
        document.getElementById("zonaProduse").style.display = "block";
        document.getElementById("zonaRezumat").style.display = "none";
        document.getElementById("titluComanda").innerText = `Comandă curentă: ${dateClient.nume}`;
        document.getElementById("detaliiClientBula").innerHTML = `📍 <b>Livrare:</b> ${dateClient.adresa || "Nespecificată"} | 📞 <b>Tel:</b> ${dateClient.telefon || "Nespecificat"}`;
        comandaCurenta = {}; document.getElementById("cautaProdus").value = "";
        construiesteInterfataProduse();
    }

    function construiesteInterfataProduse() {
        let istoricClient = istoricPreturiClienti[clientCurent] || {};
        let cheiInIstoric = Object.keys(istoricClient);
        let divIstoric = document.getElementById("sectiuneIstoric");
        let listaIstoric = document.getElementById("listaIstoric");
        let listaCatalog = document.getElementById("listaCatalog");
        listaIstoric.innerHTML = ""; listaCatalog.innerHTML = "";
        let areIstoric = false;

        catalogMarfa.forEach(prod => {
            let varianteCumparate = prod.variante.filter(v => cheiInIstoric.includes(`${prod.id}||${v.dim}`));
            if (varianteCumparate.length > 0) { areIstoric = true; listaIstoric.innerHTML += genereazaCardProdus(prod, varianteCumparate, "istoric"); }
        });
        divIstoric.style.display = areIstoric ? "block" : "none";

        catalogMarfa.forEach(prod => {
            let varianteNoi = prod.variante.filter(v => !cheiInIstoric.includes(`${prod.id}||${v.dim}`));
            if (varianteNoi.length > 0) { listaCatalog.innerHTML += genereazaCardProdus(prod, varianteNoi, "catalog"); }
        });
    }

    function genereazaCardProdus(prod, variante, tip) {
        let randuriVariante = "";
        let istoricClient = istoricPreturiClienti[clientCurent] || {};
        variante.forEach(v => {
            let cheieUnica = `${prod.id}||${v.dim}`;
            let pretDeAfisat = (tip === "istoric") ? (istoricClient[cheieUnica] || v.pret) : v.pret;
            randuriVariante += `
                <tr>
                    <td><b>${v.dim} cm</b></td>
                    <td><input type="number" step="0.01" class="input-pret-modificabil" id="pret_${tip}_${cheieUnica}" value="${pretDeAfisat}" onchange="salveazaModificareLive('${cheieUnica}', '${tip}')"> lei</td>
                    <td><input type="number" min="0" class="input-cantitate" id="cant_${tip}_${cheieUnica}" placeholder="0" onchange="salveazaModificareLive('${cheieUnica}', '${tip}')"> buc</td>
                </tr>
            `;
        });
        return `<div class="card-produs"><div><div class="produs-antet">${prod.nume}</div><table class="tabel-variante"><thead><tr><th>Dimensiune</th><th>Preț Unitar</th><th>Cantitate</th></tr></thead><tbody>${randuriVariante}</tbody></table></div></div>`;
    }

    function salveazaModificareLive(cheieUnica, tip) {
        let pretInput = document.getElementById(`pret_${tip}_${cheieUnica}`).value;
        let cantInput = document.getElementById(`cant_${tip}_${cheieUnica}`).value;
        let cantitate = parseInt(cantInput) || 0; let pret = parseFloat(pretInput) || 0;
        if (cantitate > 0) { comandaCurenta[cheieUnica] = { cantitate: cantitate, pret: pret }; } else { delete comandaCurenta[cheieUnica]; }
    }

    function filtreazaProduse() {
        let text = document.getElementById("cautaProdus").value.toLowerCase();
        let carduri = document.querySelectorAll("#listaCatalog .card-produs");
        carduri.forEach(card => {
            let nume = card.querySelector(".produs-antet").innerText.toLowerCase();
            card.style.display = nume.includes(text) ? "" : "none";
        });
    }

    function finalizareComandaCompleta() {
        let cheiComandate = Object.keys(comandaCurenta);
        if (cheiComandate.length === 0) { alert("Introdu cantitatea!"); return; }
        let dateClient = listaClientiObj.find(c => c.nume === clientCurent);
        let istoricClientCurent = istoricPreturiClienti[clientCurent] || {};

        let acum = new Date();
        let dataAfisabila = acum.toLocaleDateString('ro-RO', { day: '2-digit', month: '2-digit', year: 'numeric' }) + " " + acum.toLocaleTimeString('ro-RO', { hour: '2-digit', minute: '2-digit' });

        let textRezumat = `📋 *COMANDĂ NOUĂ CUTII*\n📅 *Data:* ${dataAfisabila}\n👤 *Client:* ${clientCurent}\n📍 *Livrare:* ${dateClient.adresa || 'Fără'}\n📞 *Tel:* ${dateClient.telefon || 'Fără'}\n-------------------------\n`;
        
        cheiComandate.forEach(cheie => {
            let [prodId, dim] = cheie.split("||");
            let pOriginal = catalogMarfa.find(p => p.id === prodId);
            let detalii = comandaCurenta[cheie];
            textRezumat += `- ${pOriginal.nume} [${dim} cm] -> *${detalii.cantitate} buc* (Preț: ${detalii.pret.toFixed(2)} lei/buc)\n`;
            istoricClientCurent[cheie] = detalii.pret;
        });
        textRezumat += `-------------------------`;

        let comandaIstoric = {
            timestamp: Date.now(),
            dataText: dataAfisabila,
            clientNume: clientCurent,
            textComplet: textRezumat
        };
        jurnalComenziTrimise.unshift(comandaIstoric);

        istoricPreturiClienti[clientCurent] = istoricClientCurent;
        salveazaInMemorie();
        randareJurnalComenzi();

        textGlobalRezumat = textRezumat;
        document.getElementById("textRezumatFinal").value = textRezumat;
        document.getElementById("zonaRezumat").style.display = "block";
        document.getElementById("zonaRezumat").scrollIntoView({ behavior: 'smooth' });
        
        alert("Comanda a fost salvată în Jurnal!");
    }

    function randareJurnalComenzi() {
        let divJurnal = document.getElementById("listaJurnalComenzi");
        if (jurnalComenziTrimise.length === 0) { divJurnal.innerHTML = "<p style='color: #718096; font-style: italic;'>Jurnal gol.</p>"; return; }

        let html = "";
        jurnalComenziTrimise.forEach((com, index) => {
            html += `
                <div class="card-comanda-veche" data-client="${com.clientNume.toLowerCase()}" data-data="${com.dataText}">
                    <div class="antet-comanda-veche">
                        <span>👤 ${com.clientNume} | 📅 ${com.dataText}</span>
                        <button class="btn-trimite-jurnal" onclick="trimiteDinJurnal(${index})">Trimite la Facturare 💬</button>
                    </div>
                    <div class="text-comanda-veche">${com.textComplet}</div>
                </div>
            `;
        });
        divJurnal.innerHTML = html;
    }

    function trimiteDinJurnal(index) {
        let com = jurnalComenziTrimise[index];
        window.open(`https://api.whatsapp.com/send?text=${encodeURIComponent(com.textComplet)}`, '_blank');
    }

    function filtreazaJurnal() {
        let textCautat = document.getElementById("cautaInJurnal").value.toLowerCase();
        let carduriComenzi = document.querySelectorAll("#listaJurnalComenzi .card-comanda-veche");
        carduriComenzi.forEach(card => {
            let client = card.getAttribute("data-client");
            let data = card.getAttribute("data-data");
            card.style.display = (client.includes(textCautat) || data.includes(textCautat)) ? "" : "none";
        });
    }

    function copiazaTextul() {
        let textarea = document.getElementById("textRezumatFinal"); textarea.select();
        navigator.clipboard.writeText(textarea.value); alert("Copiat!");
    }

    function trimitePeWhatsApp() {
        if (!textGlobalRezumat) return;
        window.open(`https://api.whatsapp.com/send?text=${encodeURIComponent(textGlobalRezumat)}`, '_blank');
    }
</script>
